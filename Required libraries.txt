Python 3.6 or higher
Recommend using Anaconda as a package manager

cv2		python -m pip install opencv-python
		conda install -c conda-forge opencv

numpy		python -m pip install numpy
		conda install -c anaconda numpy

colormap	python -m pip install colormap
		conda install -c conda-forge colormap

matplotlib	python -m pip install matplotlib
		conda install -c conda-forge matplotlib

PIL		python -m pip install Pillow
		conda install -c anaconda pillow

PyQt5		pip install PyQt5
		conda install -c anaaconda pyqt

pyqtgraph	python -m pip install pyqtgraph
		conda install -c anaconda pyqtgraph

scipy		python -m pip install scipy
		conda install -c anaconda scipy
